fib1 = 1 # Необходимые переменные
fib2 = 1 
n = input("Номер элемента ряда Фибоначчи: ")
n = int(n)
i = 0
while i < n - 2: # Выполнение "Банального алгоритма"
#
#    ПРобелы для проверки правильности
#
    fibsum = fib1 + fib2
    fib1 = fib2
    fib2 = fibsum
    i = i + 1
print("Значение этого элемента:", fib2)
